public class MSSphereRTPC : MonoBehaviour
{

    public GameObject Player;
    [Header("Centre de la Sphère = 0")]
    [Header("Bords de la Sphère  = 100")]
    [SerializeField] [ParamRef] public string insideRTPC;
    public bool DebugP = false;

    [SerializeField] private float paramMaxRange;


    private void Update()
    {

        // compute the % value
        float dist = Vector3.Distance(Player.transform.position, transform.position);
        float radius = GetComponent<CircleCollider2D>().radius;

        if (dist > radius)
        {
            //RuntimeManager.StudioSystem.setParameterByName(insideRTPC, 0);
            return;
        }
            
        float p = (dist / radius);
        float pInv = (-p + 1) * 2;

        if (pInv < paramMaxRange)
        {
            RuntimeManager.StudioSystem.setParameterByName(insideRTPC, pInv);
        }
        //Debug.Log(p);
        //RuntimeManager.StudioSystem.setParameterByName(insideRTPC, Mathf.Abs(velocity.x + velocity.y));
        if (DebugP == true)
        {
            Debug.Log(pInv);
        }
    }
}
